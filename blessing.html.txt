    <script>
        // 核心配置：1.5秒自动切 + 稳定音乐外链（亲测手机/微信可用）
        const TIP_INTERVAL = 1500;
        const MUSIC_URL = 'https://cdn.jsdelivr.net/gh/ixiaowang123/StaticResource@main/music/soft-piano.mp3';
        // 暖心提示库
        const tips = [
            "多喝水哦~ 🧊",
            "保持微笑呀 😊",
            "今天也要元气满满哦 💪",
            "记得吃水果~ 🍎",
            "保持好心情 ✨",
            "好好爱自己 ❤️",
            "我想你了 🥺",
            "梦想成真 🎯",
            "期待下一次见面 🤗",
            "前程似锦 🚀",
            "前路灿灿 🌟",
            "早点休息 😴",
            "愿所有烦恼都消失 🧹",
            "别熬夜 🛌",
            "今天过得开心嘛 🥰",
            "天冷了，多穿衣服 🧥"
        ];

        let currentIndex = 0;
        let autoTimer = null;
        let bgMusic = null;
        // 获取DOM元素
        const popup = document.getElementById('tipPopup');
        const tipText = document.getElementById('tipText');
        const pageInfo = document.getElementById('pageInfo');
        const funcBtn = document.getElementById('funcBtn');
        const completeBox = document.getElementById('completeBox');

        // 初始化音乐（强制触发播放，兼容所有浏览器）
        function initMusic() {
            bgMusic = new Audio(MUSIC_URL);
            bgMusic.loop = true; // 循环播放
            bgMusic.volume = 0.4; // 适度音量，不扰民
            // 强制播放处理，解决浏览器拦截
            bgMusic.play().catch(err => {
                // 备用播放方案，确保音乐能响
                const playEvent = () => {
                    bgMusic.play();
                    document.removeEventListener('touchstart', playEvent);
                    document.removeEventListener('click', playEvent);
                };
                document.addEventListener('touchstart', playEvent);
                document.addEventListener('click', playEvent);
            });
        }

        // 开启1.5秒自动切换
        function startAutoPlay() {
            clearAutoPlay();
            autoTimer = setInterval(() => {
                if (currentIndex < tips.length - 1) {
                    currentIndex++;
                    showCurrentTip();
                }
            }, TIP_INTERVAL);
        }

        // 清除自动切换
        function clearAutoPlay() {
            if (autoTimer) clearInterval(autoTimer);
        }

        // 显示当前提示（带淡入动画）
        function showCurrentTip() {
            tipText.textContent = tips[currentIndex];
            pageInfo.textContent = `${currentIndex+1}/${tips.length}`;
            // 最后一条停止自动切，显示完成提示
            if (currentIndex === tips.length - 1) {
                clearAutoPlay();
                funcBtn.style.display = 'none';
                completeBox.style.display = 'block';
            } else {
                funcBtn.style.display = 'flex';
                completeBox.style.display = 'none';
                startAutoPlay();
            }
        }

        // 上/下一条手动切换（切完继续1.5秒自动播）
        function prevTip() {
            if (currentIndex > 0) {
                currentIndex--;
                showCurrentTip();
            }
        }
        function nextTip() {
            if (currentIndex < tips.length - 1) {
                currentIndex++;
                showCurrentTip();
            }
        }

        // 开始所有提示（用户点击触发，无拦截）
        function startAllTip() {
            currentIndex = 0;
            initMusic(); // 初始化并播放音乐
            popup.style.display = 'flex';
            showCurrentTip();
        }

        // 关闭弹窗（暂停音乐）
        function closePopup() {
            popup.style.display = 'none';
            clearAutoPlay();
            if (bgMusic) bgMusic.pause();
        }

        // 预加载音乐，提升播放体验
        window.onload = function() {
            bgMusic = new Audio(MUSIC_URL);
            bgMusic.volume = 0.4;
        };
    </script>
